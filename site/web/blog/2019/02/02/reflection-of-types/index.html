   { 0. .. 100000000000000000. }

    #time
let reflectEach =
   instances 
   |> Seq.map(
      fun x -> 
         let props = x.GetType().GetProperties()
         let values = props |> Seq.map(fun prop -> (prop.Name, prop.GetValue(x)))
         values
   )
#time
printfn "%A" reflectEach
